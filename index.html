/** * 연차 관리 시스템 Backend (최신 프론트엔드 완벽 호환) * - 관리자 자동 승인 및 즉시 차감 지원 * - '반차', '잔여' 텍스트 인식에 따른 소수점(시간) 차감 완벽 지원 * - ID 타입 캐스팅(String)을 통한 데이터 누락/버그 원천 차단 */ const SPREADSHEET_ID = "1FGruRHR_n010jDsINPNMq2kLIWXUbgRAumBIRvWqdMc"; // 본인의 시트 ID const USER_SHEET_NAME = "회원정보"; const LEAVE_SHEET_NAME = "연차관리"; function jsonResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { let params; try { params = JSON.parse(e.postData.contents); } catch (err) { return jsonResponse({ status: 'error', message: '잘못된 데이터 형식' }); } const action = params.action; try { switch(action) { case 'register': return register(params); case 'login': return login(params); case 'requestLeave': return requestLeave(params); case 'fetchData': return fetchData(params); case 'updateStatus': return updateStatus(params); case 'deleteRequest': return deleteRequest(params); default: throw "알 수 없는 작업입니다."; } } catch (err) { return jsonResponse({ status: 'error', message: err.toString() }); } } function register(p) { const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(USER_SHEET_NAME); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === String(p.id)) throw "이미 존재하는 아이디입니다."; } const totalLeave = p.totalLeave !== undefined ? p.totalLeave : 15; sheet.appendRow([p.id, p.pw, p.name, p.phone, p.birth, p.joinDate, p.role || 'user', totalLeave, 0]); return jsonResponse({ status: 'success' }); } function getAllUsers() { const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(USER_SHEET_NAME); const data = sheet.getDataRange().getValues(); const users = []; for (let i = 1; i < data.length; i++) { users.push({ id: String(data[i][0]), name: data[i][2], role: data[i][6], total: data[i][7], used: data[i][8] }); } return users; } function login(p) { const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(USER_SHEET_NAME); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === String(p.id) && String(data[i][1]) === String(p.pw)) { const user = { id: String(data[i][0]), name: data[i][2], role: data[i][6], total: data[i][7], used: data[i][8] }; let res = { status: 'success', user: user, requests: getRequests() }; if (String(user.role).includes('admin')) res.allUsers = getAllUsers(); return jsonResponse(res); } } throw "아이디 또는 비밀번호가 일치하지 않습니다."; } function getRequests() { const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(LEAVE_SHEET_NAME); const data = sheet.getDataRange().getValues(); const list = []; for (let i = 1; i < data.length; i++) { list.push({ id: String(data[i][0]), requestDate: data[i][1], userId: String(data[i][2]), userName: data[i][3], userRole: data[i][4], leaveDate: data[i][5], type: data[i][6], reason: data[i][7], handover: data[i][8], status: data[i][9] }); } return list.reverse(); } function fetchData(p) { const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(USER_SHEET_NAME); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === String(p.id)) { const user = { id: String(data[i][0]), name: data[i][2], role: data[i][6], total: data[i][7], used: data[i][8] }; let res = { status: 'success', user: user, requests: getRequests() }; if (String(user.role).includes('admin')) res.allUsers = getAllUsers(); return jsonResponse(res); } } throw "사용자를 찾을 수 없습니다."; } // 🚀 백엔드 시간제 연차 차감 로직 (소수점 처리) function getDeductionDays(type) { if (!type) return 0; const t = String(type).toLowerCase(); if (t.includes('무급')) return 0; if (t.includes('잔여')) return 3 / 8; // 3시간 차감 (0.375일) - 우선 체크 if (t.includes('반차')) return 5 / 8; // 5시간 차감 (0.625일) return 1; // 일반 연차 (1일) } function requestLeave(p) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const leaveSheet = ss.getSheetByName(LEAVE_SHEET_NAME); const status = p.status ? p.status : "대기"; leaveSheet.appendRow([ Date.now().toString(), p.requestDate, p.userId, p.userName, p.userRole, p.leaveDate, p.type, p.reason, p.handover, status ]); // 관리자 신청 등 즉시 '최종 승인'되는 경우 바로 수량 삭감 if (status === '최종 승인') { // 프론트에서 받은 deduction 값이 있으면 우선 사용, 없으면 백엔드에서 계산 const diff = p.deduction !== undefined ? parseFloat(p.deduction) : getDeductionDays(p.type); const userSheet = ss.getSheetByName(USER_SHEET_NAME); const userData = userSheet.getDataRange().getValues(); for (let j = 1; j < userData.length; j++) { if (String(userData[j][0]) === String(p.userId)) { let currentUsed = parseFloat(userData[j][8] || 0); userSheet.getRange(j + 1, 9).setValue(currentUsed + diff); break; } } } return jsonResponse({ status: 'success' }); } function updateStatus(p) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const leaveSheet = ss.getSheetByName(LEAVE_SHEET_NAME); const leaveData = leaveSheet.getDataRange().getValues(); let rowIndex = -1; let targetReq = null; for (let i = 1; i < leaveData.length; i++) { if (String(leaveData[i][0]) === String(p.requestId)) { rowIndex = i + 1; targetReq = { userId: String(leaveData[i][2]), type: leaveData[i][6], status: leaveData[i][9] }; break; } } if (rowIndex === -1) throw "내역 없음"; const oldStatus = targetReq.status; const newStatus = p.status; // 상태 업데이트 leaveSheet.getRange(rowIndex, 10).setValue(newStatus); const diff = getDeductionDays(targetReq.type); const userSheet = ss.getSheetByName(USER_SHEET_NAME); const userData = userSheet.getDataRange().getValues(); for (let j = 1; j < userData.length; j++) { if (String(userData[j][0]) === String(targetReq.userId)) { let currentUsed = parseFloat(userData[j][8] || 0); if (oldStatus !== '최종 승인' && newStatus === '최종 승인') { currentUsed += diff; } else if (oldStatus === '최종 승인' && newStatus !== '최종 승인') { currentUsed -= diff; } userSheet.getRange(j + 1, 9).setValue(currentUsed); break; } } return jsonResponse({ status: 'success' }); } function deleteRequest(p) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const leaveSheet = ss.getSheetByName(LEAVE_SHEET_NAME); const leaveData = leaveSheet.getDataRange().getValues(); let rowIndex = -1; let targetReq = null; for (let i = 1; i < leaveData.length; i++) { if (String(leaveData[i][0]) === String(p.requestId)) { rowIndex = i + 1; targetReq = { userId: String(leaveData[i][2]), type: leaveData[i][6], status: leaveData[i][9] }; break; } } if (rowIndex === -1) throw "내역 없음"; // 삭제하는 내역이 '최종 승인' 상태였다면 연차를 다시 복구 if (targetReq.status === '최종 승인') { const diff = getDeductionDays(targetReq.type); const userSheet = ss.getSheetByName(USER_SHEET_NAME); const userData = userSheet.getDataRange().getValues(); for (let j = 1; j < userData.length; j++) { if (String(userData[j][0]) === String(targetReq.userId)) { let currentUsed = parseFloat(userData[j][8] || 0); userSheet.getRange(j + 1, 9).setValue(currentUsed - diff); break; } } } leaveSheet.deleteRow(rowIndex); return jsonResponse({ status: 'success' }); }
